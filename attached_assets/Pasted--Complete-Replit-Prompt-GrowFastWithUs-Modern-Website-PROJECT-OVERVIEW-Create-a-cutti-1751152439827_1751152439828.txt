### Dark/Light Mode Implementation:
```javascript
// Theme Toggle Component
const ThemeToggle = () => {
  const [isDark, setIsDark] = useState(false)
  
  const toggleTheme = () => {
    setIsDark(!isDark)
    document.documentElement.classList.toggle('dark')
  }
  
  return (
    <motion.button
      className="glass-button fixed top-4 right-4 z-50"
      onClick={toggleTheme}
      whileHover={{ scale: 1.1 }}
      whileTap={{ scale: 0.9 }}
    >
      <motion.div
        animate={{ rotate: isDark ? 180 : 0 }}
        transition={{ duration: 0.5 }}
      >
        {isDark ? <Sun /> : <Moon />}
      </motion.div>
    </motion.button>
  )
}
```
